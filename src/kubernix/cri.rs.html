22/// Environment variable name for the CRI socket endpoint.
23pub const RUNTIME_ENV: &str = "CONTAINER_RUNTIME_ENDPOINT";
24
25/// Maximum usable bytes in a Unix socket path (108 - 1 for null terminator).
26pub const MAX_SOCKET_PATH_LEN: usize = 107;
27
28/// Simple CRI socket abstraction.
29pub struct CriSocket(PathBuf);
30
31impl Display for CriSocket {
32    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
33        write!(f, "{}", self.0.display())
34    }
35}
36
37impl CriSocket {
38    /// Create a new CRI socket, validating the path length against Unix limits.
39    pub fn new(path: PathBuf) -> Result<CriSocket> {
40        if path.display().to_string().len() > MAX_SOCKET_PATH_LEN {
41            bail!("Socket path '{}' is too long", path.display())
42        }
43        Ok(CriSocket(path))
44    }
45
46    /// Format the socket path as a `unix://` URI for CRI clients.
47    pub fn to_socket_string(&self) -> String {
48        format!("unix://{}", self.0.display())
49    }
50}
51
52/// Return the CRI socket for the configured runtime and node.
53pub fn cri_socket(config: &Config, network: &Network, node: u8) -> Result<CriSocket> {
54    match config.cri_runtime() {
55        CriRuntime::Crio => Crio::socket(config, network, node),
56        CriRuntime::Containerd => Containerd::socket(config, network, node),
57    }
58}

60/// Write the CNI bridge network configuration for a node.
61pub fn write_cni_config(
62    cni_conf_dir: &Path,
63    node_name: &str,
64    node: u8,
65    network: &Network,
66) -> Result<()> {
67    let cidr = network
68        .pod_cidrs()
69        .get(node as usize)
70        .with_context(|| format!("Unable to find CIDR for {}", node_name))?;
71    fs::write(
72        cni_conf_dir.join("10-bridge.json"),
73        to_string_pretty(&json!({
74            "cniVersion": "0.3.1",
75            "name": format!("kubernix-{}", node_name),
76            "type": "bridge",
77            "bridge": format!("{}.{}", Network::INTERFACE_PREFIX, node),
78            "isGateway": true,
79            "ipMasq": true,
80            "hairpinMode": true,
81            "ipam": {
82                "type": "host-local",
83                "routes": [{ "dst": "0.0.0.0/0" }],
84                "ranges": [[{ "subnet": cidr }]]
85            }
86        }))?,
87    )?;
88    Ok(())
89}

91/// Remove all containers and pods via crictl.
92pub fn remove_all_containers(label: &str, socket: &CriSocket, node_name: &str) -> Result<()> {
93    debug!("Removing all {} workloads on {}", label, node_name);
94
95    let output = Command::new("crictl")
96        .env(RUNTIME_ENV, socket.to_socket_string())
97        .arg("pods")
98        .arg("-q")
99        .output()?;
100    let stdout = String::from_utf8(output.stdout)?;
101    if !output.status.success() {
102        debug!("crictl pods stdout ({}): {}", node_name, stdout);
103        debug!(
104            "crictl pods stderr ({}): {}",
105            node_name,
106            String::from_utf8(output.stderr)?
107        );
108        bail!("crictl pods command failed ({})", node_name);
109    }
110
111    for x in stdout.lines() {
112        debug!("Removing pod {} on {}", x, node_name);
113        let output = Command::new("crictl")
114            .env(RUNTIME_ENV, socket.to_socket_string())
115            .arg("rmp")
116            .arg("-f")
117            .arg(x)
118            .output()?;
119        if !output.status.success() {
120            debug!("crictl rmp ({}): {:?}", node_name, output);
121            bail!("crictl rmp command failed ({})", node_name);
122        }
123    }
124
125    debug!("All workloads removed on {}", node_name);
126    Ok(())
127}
