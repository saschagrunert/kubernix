44    pub fn new(config: &Config, pki: &Pki) -> Result<KubeConfig> {
45        // Create the target dir
46        let dir = config.root().join("kubeconfig");
47
48        if dir.exists() {
49            info!("Kubeconfig directory already exists, skipping generation");
50
51            let kubelets = pki
52                .kubelets()
53                .iter()
54                .map(|i| Self::target_config(&dir, i))
55                .collect();
56
57            Ok(KubeConfig {
58                kubelets,
59                proxy: Self::target_config(&dir, pki.proxy()),
60                controller_manager: Self::target_config(&dir, pki.controller_manager()),
61                scheduler: Self::target_config(&dir, pki.scheduler()),
62                admin: Self::target_config(&dir, pki.admin()),
63            })
64        } else {
65            info!("Creating kubeconfigs");
66            create_dir_all(&dir)?;
67
68            let kubelets = pki
69                .kubelets()
70                .iter()
71                .map(|x| Self::setup_kubeconfig(&dir, x, pki.ca().cert()))
72                .collect::<Result<Vec<_>, _>>()?;
73
74            Ok(KubeConfig {
75                kubelets,
76                proxy: Self::setup_kubeconfig(&dir, pki.proxy(), pki.ca().cert())?,
77                controller_manager: Self::setup_kubeconfig(
78                    &dir,
79                    pki.controller_manager(),
80                    pki.ca().cert(),
81                )?,
82                scheduler: Self::setup_kubeconfig(&dir, pki.scheduler(), pki.ca().cert())?,
83                admin: Self::setup_kubeconfig(&dir, pki.admin(), pki.ca().cert())?,
84            })
85        }
86    }

88    fn setup_kubeconfig(dir: &Path, identity: &Identity, ca: &Path) -> Result<PathBuf> {
89        debug!("Creating kubeconfig for {}", identity.name());
90        let kubeconfig = Self::target_config(dir, identity);
91
92        let embed_certs = "--embed-certs=true";
93        let cluster = "kubernetes";
94        let kubectl = Kubectl::new(&kubeconfig);
95        kubectl.config(&[
96            "set-cluster",
97            cluster,
98            &format!("--certificate-authority={}", ca.display()),
99            &format!("--server=https://{}:6443", &Ipv4Addr::LOCALHOST),
100            embed_certs,
101        ])?;
102
103        kubectl.config(&[
104            "set-credentials",
105            identity.user(),
106            &format!("--client-certificate={}", identity.cert().display()),
107            &format!("--client-key={}", identity.key().display()),
108            embed_certs,
109        ])?;
110
111        let context = "kubernix";
112        kubectl.config(&[
113            "set-context",
114            context,
115            &format!("--cluster={}", cluster),
116            &format!("--user={}", identity.user()),
117        ])?;
118
119        kubectl.config(&["use-context", context])?;
120
121        // Adapt file permissions
122        let file = File::open(&kubeconfig).context("unable to open kubeconfig")?;
123        fchmod(
124            &file,
125            Mode::from_bits(0o600).ok_or_else(|| format_err!("unable to get mode bits"))?,
126        )
127        .context("unable to set kubeconfig permissions")?;
128
129        debug!("Kubeconfig created for {}", identity.name());
130        Ok(kubeconfig)
131    }
132
133    fn target_config(dir: &Path, identity: &Identity) -> PathBuf {
134        dir.join(format!("{}.kubeconfig", identity.name()))
135    }
136}
