17
18impl Kubelet {
19    pub fn start(
20        config: &Config,
21        node: u8,
22        network: &Network,
23        pki: &Pki,
24        kubeconfig: &KubeConfig,
25    ) -> ProcessState {
26        let node_name = Node::name(config, network, node);
27        const KUBELET: &str = "kubelet";
28
29        let dir = config.root().join(KUBELET).join(&node_name);
30        let root_dir = dir.join("run");
31        // pod-resources/<pid> is the longest socket path kubelet creates
32        if root_dir.display().to_string().len() + "pod-resources/1234567890".len()
33            > MAX_SOCKET_PATH_LEN
34        {
35            bail!(
36                "Kubelet run path '{}' is too long for unix sockets",
37                root_dir.display()
38            )
39        }
40
41        create_dir_all(&dir)?;
42
43        let identity = pki
44            .kubelets()
45            .get(node as usize)
46            .with_context(|| format!("Unable to retrieve kubelet identity for {}", node_name))?;
47
48        let yml = format!(
49            include_str!("assets/kubelet.yml"),
50            ca = pki.ca().cert().display(),
51            dns = network.dns()?,
52            cidr = network
53                .crio_cidrs()
54                .get(node as usize)
55                .context("Unable to retrieve kubelet CIDR")?,
56            cert = identity.cert().display(),
57            key = identity.key().display(),
58            port = 11250 + u16::from(node),
59            healthzPort = 12250 + u16::from(node),
60        );
61        let cfg = dir.join("config.yml");
62
63        if !cfg.exists() {
64            fs::write(&cfg, yml)?;
65        }
66
67        let args = &[
68            &format!("--config={}", cfg.display()),
69            &format!("--root-dir={}", root_dir.display()),
70            &format!(
71                "--container-runtime-endpoint={}",
72                Crio::socket(config, network, node)?.to_socket_string(),
73            ),
74            &format!(
75                "--kubeconfig={}",
76                kubeconfig
77                    .kubelets()
78                    .get(node as usize)
79                    .with_context(|| format!(
80                        "Unable to retrieve kubelet config for {}",
81                        node_name
82                    ))?
83                    .display()
84            ),
85            "--v=2",
86        ];
87
88        let mut process = if config.multi_node() {
89            // Run inside a container
90            let arg_hostname = &format!("--hostname-override={}", node_name);
91            let mut modargs: Vec<&str> = vec![arg_hostname];
92            modargs.extend(args);
93            Container::exec(
94                config,
95                &dir,
96                &format!("Kubelet {}", node_name),
97                KUBELET,
98                &node_name,
99                &modargs,
100            )?
101        } else {
102            // Run as usual process
103            Process::start(&dir, "Kubelet", KUBELET, args)?
104        };
105        process.wait_ready("Successfully registered node")?;
106        Ok(Box::new(Self { process }))
107    }
108}
109
110impl Stoppable for Kubelet {
111    fn stop(&mut self) -> Result<()> {
112        self.process.stop()
113    }
