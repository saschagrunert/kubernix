17    /// Build the base image used for the nodes
18    pub fn build(config: &Config) -> Result<()> {
19        // Verify that the provided runtime exists
20        System::find_executable(config.container_runtime())?;
21
22        // Write the policy file
23        let policy_json = Self::policy_json(config);
24        fs::write(&policy_json, include_str!("assets/policy.json"))?;
25
26        // Nothing needs to be done on single node runs or root users
27        if !config.multi_node() {
28            return Ok(());
29        }
30
31        // Build the base container image
32        info!("Building base container image '{}'", DEFAULT_IMAGE);
33
34        // Prepare the Dockerfile: use a custom one if provided, otherwise
35        // fall back to the embedded default.
36        let file = config.root().join("Dockerfile");
37        if !file.exists() {
38            if let Some(custom) = config.dockerfile() {
39                debug!("Using custom Dockerfile '{}'", custom.display());
40                fs::copy(custom, &file)?;
41            } else {
42                fs::write(
43                    &file,
44                    format!(
45                        include_str!("assets/Dockerfile"),
46                        nix = Nix::DIR,
47                        root = DEFAULT_ROOT
48                    ),
49                )?;
50            }
51        }
52
53        // Exclude .git from the container build context since the
54        // runtime nix directory uses a standalone git repo that the
55        // container image does not need.
56        let dockerignore = config.root().join(".dockerignore");
57        if !dockerignore.exists() {
58            fs::write(&dockerignore, "nix/.git\n")?;
59        }
60
61        // Prepare the arguments
62        let mut args = if Podman::is_configured(config) {
63            Podman::build_args(config, &policy_json)?
64        } else {
65            vec!["build".into()]
66        };
67        args.extend(vec![format!("-t={}", DEFAULT_IMAGE), ".".into()]);
68        trace!("Container runtime build args: {:?}", args);
69
70        // Run the build
71        debug!("Running container runtime with args: {}", args.join(" "));
72        let status = Command::new(config.container_runtime())
73            .current_dir(config.root())
74            .args(args)
75            .stderr(Self::stdio(config))
76            .stdout(Self::stdio(config))
77            .status()?;
78        if !status.success() {
79            bail!(
80                "Unable to build container base image '{}' using '{}' (exit: {})",
81                DEFAULT_IMAGE,
82                config.container_runtime(),
83                status,
84            );
85        }
86
87        info!("Container base image built");
88        Ok(())
89    }

96    /// Start a new container based process
97    pub fn start(
98        config: &Config,
99        dir: &Path,
100        identifier: &str,
101        process_name: &str,
102        container_name: &str,
103        args: &[&str],
104    ) -> Result<Process> {
105        // Cleanup possible containers
106        Self::remove(config, container_name);
107
108        // Prepare the arguments
109        let arg_hostname = &format!("--hostname={}", container_name);
110        let arg_name = &format!("--name={}", Self::prefixed_container_name(container_name));
111        let arg_volume_root = &Self::volume_arg(config.root().display());
112        let mut args_vec = vec![
113            "run",
114            "--net=host",
115            "--privileged",
116            "--rm",
117            "--cgroupns=host",
118            arg_hostname,
119            arg_name,
120            arg_volume_root,
121        ];
122
123        // Podman specific arguments
124        let podman_args = Podman::default_args(config)?;
125        if Podman::is_configured(config) {
126            args_vec.extend(podman_args.iter().map(|x| x.as_str()).collect::<Vec<_>>())
127        }
128
129        // Mount /dev/mapper if available
130        let dev_mapper = PathBuf::from("/").join("dev").join("mapper");
131        let arg_volume_dev_mapper = &Self::volume_arg(dev_mapper.display());
132        if dev_mapper.exists() {
133            args_vec.push(arg_volume_dev_mapper);
134        }
135
136        // Add the process and the user provided args
137        args_vec.extend(&[DEFAULT_IMAGE, process_name]);
138        args_vec.extend(args);
139
140        // Start the process
141        trace!("Container runtime start args: {:?}", args_vec);
142        Process::start(dir, identifier, config.container_runtime(), &args_vec)
143    }
144
145    fn volume_arg<T: Display>(volume: T) -> String {
146        format!("--volume={v}:{v}", v = volume)
147    }

149    /// Exec a command on a container instance
150    pub fn exec(
151        config: &Config,
152        dir: &Path,
153        identifier: &str,
154        process_name: &str,
155        container_name: &str,
156        args: &[&str],
157    ) -> Result<Process> {
158        // Prepare the args
159        let mut args_vec = vec![];
160
161        let podman_args = Podman::default_args(config)?;
162        if Podman::is_configured(config) {
163            args_vec.extend(podman_args.iter().map(|x| x.as_str()).collect::<Vec<_>>())
164        }
165
166        let name = Self::prefixed_container_name(container_name);
167        let flake_ref = format!("path:{}", DEFAULT_ROOT);
168        let mut cmd_parts = vec![process_name.to_string()];
169        cmd_parts.extend(args.iter().map(|a| a.to_string()));
170        let run_cmd = cmd_parts.join(" ");
171        args_vec.extend(vec![
172            "exec",
173            &name,
174            "nix",
175            "develop",
176            &flake_ref,
177            "--no-update-lock-file",
178            "--command",
179            "bash",
180            "-c",
181            &run_cmd,
182        ]);
183
184        // Run as usual process
185        trace!("Container runtime exec args: {:?}", args_vec);
186        Process::start(dir, identifier, config.container_runtime(), &args_vec)
187    }

189    /// Remove the provided (maybe running) container.
190    /// Failures are logged but not propagated since the container may not
191    /// exist yet (first run).
192    fn remove(config: &Config, name: &str) {
193        let prefixed = Self::prefixed_container_name(name);
194        match Command::new(config.container_runtime())
195            .arg("rm")
196            .arg("-f")
197            .arg(&prefixed)
198            .stderr(Stdio::null())
199            .stdout(Stdio::null())
200            .status()
201        {
202            Ok(status) if !status.success() => {
203                debug!("Container '{}' removal exited with {}", prefixed, status);
204            }
205            Err(e) => {
206                debug!("Failed to run container rm for '{}': {}", prefixed, e);
207            }
208            _ => {}
209        }
210    }
211
212    /// Retrieve a stdio for the provided config log level
213    fn stdio(config: &Config) -> Stdio {
214        if config.log_level() > LevelFilter::Info {
215            Stdio::inherit()
216        } else {
217            Stdio::null()
218        }
219    }

221    /// Retrieve a prefixed container name
222    fn prefixed_container_name(name: &str) -> String {
