16impl Container {
17    /// Build the base image used for the nodes
18    pub fn build(config: &Config) -> Result<()> {
19        // Verify that the provided runtime exists
20        System::find_executable(config.container_runtime())?;
21
22        // Write the policy file
23        let policy_json = Self::policy_json(config);
24        fs::write(&policy_json, include_str!("assets/policy.json"))?;
25
26        // Nothing needs to be done on single node runs or root users
27        if !config.multi_node() {
28            return Ok(());
29        }
30
31        // Build the base container image
32        info!("Building base container image '{}'", DEFAULT_IMAGE);
33
34        // Prepare the Dockerfile
35        let file = config.root().join("Dockerfile");
36        if !file.exists() {
37            fs::write(
38                &file,
39                format!(
40                    include_str!("assets/Dockerfile"),
41                    nix = Nix::DIR,
42                    root = DEFAULT_ROOT
43                ),
44            )?;
45        }
46
47        // Prepare the arguments
48        let mut args = if Podman::is_configured(config) {
49            Podman::build_args(config, &policy_json)?
50        } else {
51            vec!["build".into()]
52        };
53        args.extend(vec![format!("-t={}", DEFAULT_IMAGE), ".".into()]);
54        trace!("Container runtime build args: {:?}", args);
55
56        // Run the build
57        debug!("Running container runtime with args: {}", args.join(" "));
58        let status = Command::new(config.container_runtime())
59            .current_dir(config.root())
60            .args(args)
61            .stderr(Self::stdio(config))
62            .stdout(Self::stdio(config))
63            .status()?;
64        if !status.success() {
65            bail!("Unable to build container base image");
66        }
67
68        info!("Container base image built");
69        Ok(())
70    }

72    /// Retrieve the default signature policy file location
73    pub fn policy_json(config: &Config) -> PathBuf {
74        config.root().join("policy.json")
75    }

77    /// Start a new container based process
78    pub fn start(
79        config: &Config,
80        dir: &Path,
81        identifier: &str,
82        process_name: &str,
83        container_name: &str,
84        args: &[&str],
85    ) -> Result<Process> {
86        // Cleanup possible containers
87        Self::remove(config, container_name)?;
88
89        // Prepare the arguments
90        let arg_hostname = &format!("--hostname={}", container_name);
91        let arg_name = &format!("--name={}", Self::prefixed_container_name(container_name));
92        let arg_volume_root = &Self::volume_arg(config.root().display());
93        let mut args_vec = vec![
94            "run",
95            "--net=host",
96            "--privileged",
97            "--rm",
98            "--cgroupns=host",
99            arg_hostname,
100            arg_name,
101            arg_volume_root,
102        ];
103
104        // Podman specific arguments
105        let podman_args = Podman::default_args(config)?;
106        if Podman::is_configured(config) {
107            args_vec.extend(podman_args.iter().map(|x| x.as_str()).collect::<Vec<_>>())
108        }
109
110        // Mount /dev/mapper if available
111        let dev_mapper = PathBuf::from("/").join("dev").join("mapper");
112        let arg_volume_dev_mapper = &Self::volume_arg(dev_mapper.display());
113        if dev_mapper.exists() {
114            args_vec.push(arg_volume_dev_mapper);
115        }
116
117        // Add the process and the user provided args
118        args_vec.extend(&[DEFAULT_IMAGE, process_name]);
119        args_vec.extend(args);
120
121        // Start the process
122        trace!("Container runtime start args: {:?}", args_vec);
123        Process::start(dir, identifier, config.container_runtime(), &args_vec)
124    }
125
126    fn volume_arg<T: Display>(volume: T) -> String {
127        format!("--volume={v}:{v}", v = volume)
128    }
129
130    /// Exec a command on a container instance
131    pub fn exec(
132        config: &Config,
133        dir: &Path,
134        identifier: &str,
135        process_name: &str,
136        container_name: &str,
137        args: &[&str],
138    ) -> Result<Process> {
139        // Prepare the args
140        let mut args_vec = vec![];
141
142        let podman_args = Podman::default_args(config)?;
143        if Podman::is_configured(config) {
144            args_vec.extend(podman_args.iter().map(|x| x.as_str()).collect::<Vec<_>>())
145        }
146
147        let name = Self::prefixed_container_name(container_name);
148        let nix_dir = format!("{}/default.nix", DEFAULT_ROOT);
149        let mut cmd_parts = vec![process_name.to_string()];
150        cmd_parts.extend(args.iter().map(|a| a.to_string()));
151        let run_cmd = cmd_parts.join(" ");
152        args_vec.extend(vec![
153            "exec",
154            &name,
155            "nix-shell",
156            &nix_dir,
157            "--run",
158            &run_cmd,
159        ]);
160
161        // Run as usual process
162        trace!("Container runtime exec args: {:?}", args_vec);
163        Process::start(dir, identifier, config.container_runtime(), &args_vec)
164    }
165
166    /// Remove the provided (maybe running) container
167    fn remove(config: &Config, name: &str) -> Result<()> {
168        Command::new(config.container_runtime())
169            .arg("rm")
170            .arg("-f")
171            .arg(Self::prefixed_container_name(name))
172            .stderr(Stdio::null())
173            .stdout(Stdio::null())
174            .status()?;
175        Ok(())
176    }
177
178    /// Retrieve a stdio for the provided config log level
179    fn stdio(config: &Config) -> Stdio {
180        if config.log_level() > LevelFilter::Info {
181            Stdio::inherit()
182        } else {
183            Stdio::null()
184        }
185    }
186
187    /// Retrieve a prefixed container name
188    fn prefixed_container_name(name: &str) -> String {
189        format!("{}-{}", DEFAULT_ROOT, name)
190    }
