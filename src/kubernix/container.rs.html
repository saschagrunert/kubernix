1//! Multi-node container management.
2//!
3//! Handles building the base container image and running cluster
4//! components inside containers for multi-node setups. Each container
5//! shares the host network namespace and mounts the runtime root directory.
6
7use crate::{Config, nix::Nix, podman::Podman, process::Process, system::System};
8use anyhow::{Result, bail};
9use log::{LevelFilter, debug, info, trace};
10use std::{
11    fmt::Display,
12    fs,
13    path::{Path, PathBuf},
14    process::{Command, Stdio},
15};
16
17const DEFAULT_IMAGE: &str = "kubernix:base";
18const DEFAULT_ROOT: &str = "kubernix";
19
20/// Provides container image building and process execution for multi-node clusters.
21pub struct Container;
22
23impl Container {
24    /// Build the base image used for the nodes
25    pub fn build(config: &Config) -> Result<()> {
26        // Verify that the provided runtime exists
27        System::find_executable(config.container_runtime())?;
28
29        // Write the policy file
30        let policy_json = Self::policy_json(config);
31        fs::write(&policy_json, include_str!("assets/policy.json"))?;
32
33        // Nothing needs to be done on single node runs or root users
34        if !config.multi_node() {
35            return Ok(());
36        }
37
38        // Skip rebuild if the image already exists
39        if Self::image_exists(config) {
40            info!(
41                "Container image '{}' already exists, skipping build",
42                DEFAULT_IMAGE
43            );
44            return Ok(());
45        }
46
47        // Build the base container image
48        info!("Building base container image '{}'", DEFAULT_IMAGE);
49
50        // Prepare the Dockerfile: use a custom one if provided, otherwise
51        // fall back to the embedded default.
52        let file = config.root().join("Dockerfile");
53        if !file.exists() {
54            if let Some(custom) = config.dockerfile() {
55                debug!("Using custom Dockerfile '{}'", custom.display());
56                fs::copy(custom, &file)?;
57            } else {
58                fs::write(
59                    &file,
60                    format!(
61                        include_str!("assets/Dockerfile"),
62                        nix = Nix::DIR,
63                        root = DEFAULT_ROOT
64                    ),
65                )?;
66            }
67        }
68
69        // Exclude .git from the container build context since the
70        // runtime nix directory uses a standalone git repo that the
71        // container image does not need.
72        let dockerignore = config.root().join(".dockerignore");
73        if !dockerignore.exists() {
74            fs::write(&dockerignore, "nix/.git\n")?;
75        }
76
77        // Prepare the arguments
78        let mut args = if Podman::is_configured(config) {
79            Podman::build_args(config, &policy_json)?
80        } else {
81            vec!["build".into()]
82        };
83        args.extend(vec![format!("-t={}", DEFAULT_IMAGE), ".".into()]);
84        trace!("Container runtime build args: {:?}", args);
85
86        // Run the build
87        debug!("Running container runtime with args: {}", args.join(" "));
88        let status = Command::new(config.container_runtime())
89            .current_dir(config.root())
90            .args(args)
91            .stderr(Self::stdio(config))
92            .stdout(Self::stdio(config))
93            .status()?;
94        if !status.success() {
95            bail!(
96                "Unable to build container base image '{}' using '{}' (exit: {})",
97                DEFAULT_IMAGE,
98                config.container_runtime(),
99                status,
100            );
101        }
102
103        info!("Container base image built");
104        Ok(())
105    }
106
107    /// Retrieve the default signature policy file location
108    pub fn policy_json(config: &Config) -> PathBuf {
109        config.root().join("policy.json")
110    }
111
112    /// Start a new container based process
113    pub fn start(
114        config: &Config,
115        dir: &Path,
116        identifier: &str,
117        process_name: &str,
118        container_name: &str,
119        args: &[&str],
120    ) -> Result<Process> {
121        // Cleanup possible containers
122        Self::remove(config, container_name);
123
124        // Prepare the arguments
125        let arg_hostname = &format!("--hostname={}", container_name);
126        let arg_name = &format!("--name={}", Self::prefixed_container_name(container_name));
127        let arg_volume_root = &Self::volume_arg(config.root().display());
128        let mut args_vec = vec![
129            "run",
130            "--net=host",
131            "--privileged",
132            "--rm",
133            "--cgroupns=host",
134            arg_hostname,
135            arg_name,
136            arg_volume_root,
137        ];
138
139        // Podman specific arguments
140        let podman_args = Podman::default_args(config)?;
141        if Podman::is_configured(config) {
142            args_vec.extend(podman_args.iter().map(|x| x.as_str()).collect::<Vec<_>>())
143        }
144
145        // Mount /dev/mapper if available
146        let dev_mapper = PathBuf::from("/").join("dev").join("mapper");
147        let arg_volume_dev_mapper = &Self::volume_arg(dev_mapper.display());
148        if dev_mapper.exists() {
149            args_vec.push(arg_volume_dev_mapper);
150        }
151
152        // Add the process and the user provided args
153        args_vec.extend(&[DEFAULT_IMAGE, process_name]);
154        args_vec.extend(args);
155
156        // Start the process
157        trace!("Container runtime start args: {:?}", args_vec);
158        Process::start(dir, identifier, config.container_runtime(), &args_vec)
159    }
160
161    fn volume_arg<T: Display>(volume: T) -> String {
162        format!("--volume={v}:{v}", v = volume)
163    }
164
165    /// Exec a command on a container instance
166    pub fn exec(
167        config: &Config,
168        dir: &Path,
169        identifier: &str,
170        process_name: &str,
171        container_name: &str,
172        args: &[&str],
173    ) -> Result<Process> {
174        // Prepare the args
175        let mut args_vec = vec![];
176
177        let podman_args = Podman::default_args(config)?;
178        if Podman::is_configured(config) {
179            args_vec.extend(podman_args.iter().map(|x| x.as_str()).collect::<Vec<_>>())
180        }
181
182        let name = Self::prefixed_container_name(container_name);
183        let flake_ref = format!("path:{}", DEFAULT_ROOT);
184        let mut cmd_parts = vec![process_name.to_string()];
185        cmd_parts.extend(args.iter().map(|a| a.to_string()));
186        let run_cmd = cmd_parts.join(" ");
187        args_vec.extend(vec![
188            "exec",
189            &name,
190            "nix",
191            "develop",
192            &flake_ref,
193            "--no-update-lock-file",
194            "--command",
195            "bash",
196            "-c",
197            &run_cmd,
198        ]);
199
200        // Run as usual process
201        trace!("Container runtime exec args: {:?}", args_vec);
202        Process::start(dir, identifier, config.container_runtime(), &args_vec)
203    }

205    /// Remove the provided (maybe running) container.
206    /// Failures are logged but not propagated since the container may not
207    /// exist yet (first run).
208    fn remove(config: &Config, name: &str) {
209        let prefixed = Self::prefixed_container_name(name);
210        match Command::new(config.container_runtime())
211            .arg("rm")
212            .arg("-f")
213            .arg(&prefixed)
214            .stderr(Stdio::null())
215            .stdout(Stdio::null())
216            .status()
217        {
218            Ok(status) if !status.success() => {
219                debug!("Container '{}' removal exited with {}", prefixed, status);
220            }
221            Err(e) => {
222                debug!("Failed to run container rm for '{}': {}", prefixed, e);
223            }
224            _ => {}
225        }
226    }
227
228    /// Retrieve a stdio for the provided config log level
229    fn stdio(config: &Config) -> Stdio {
230        if config.log_level() > LevelFilter::Info {
231            Stdio::inherit()
232        } else {
233            Stdio::null()
234        }
235    }
236
237    /// Check whether the base container image already exists.
238    /// Uses `image inspect` which works for both podman and docker.
239    fn image_exists(config: &Config) -> bool {
240        Command::new(config.container_runtime())
241            .args(["image", "inspect", DEFAULT_IMAGE])
242            .stderr(Stdio::null())
243            .stdout(Stdio::null())
244            .status()
245            .is_ok_and(|s| s.success())
246    }
247
248    /// Retrieve a prefixed container name
249    fn prefixed_container_name(name: &str) -> String {
250        format!("{}-{}", DEFAULT_ROOT, name)
251    }
