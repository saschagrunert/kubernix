31/// Write `content` to `path` only if the file does not exist or its
32/// current contents differ. Avoids unnecessary filesystem writes and
33/// inode updates on warm restarts.
34pub(crate) fn write_if_changed(path: &std::path::Path, content: &str) -> anyhow::Result<()> {
35    if path.exists() {
36        if let Ok(existing) = std::fs::read_to_string(path) {
37            if existing == content {
38                return Ok(());
39            }
40        }
41    }
42    std::fs::write(path, content)?;
43    Ok(())
44}

97    /// Start kubernix by consuming the provided configuration
98    pub fn start(mut config: Config) -> Result<()> {
99        Self::prepare_env(&mut config)?;
100
101        // Bootstrap if we're not inside a nix shell
102        if Nix::is_active() {
103            Self::bootstrap_cluster(config)
104        } else {
105            Nix::bootstrap(config)
106        }
107    }

109    /// Spawn a new shell into the provided configuration environment
110    pub fn new_shell(mut config: Config) -> Result<()> {
111        Self::prepare_env(&mut config)?;
112
113        info!(
114            "Spawning new kubernix shell in: '{}'",
115            config.root().display()
116        );
117
118        let env_file = Self::env_file(&config);
119        if !env_file.exists() {
120            bail!(
121                "Necessary environment file '{}' does not exist",
122                env_file.display()
123            )
124        }
125
126        let shell_cmd = format!(". {} && exec {}", env_file.display(), config.shell_ok()?);
127        Nix::run(&config, &["bash", "-c", &format!("'{}'", shell_cmd)])?;
128
129        info!("Bye, leaving the Kubernix environment");
130        Ok(())
131    }

133    /// Prepare the environment based on the provided config
134    fn prepare_env(config: &mut Config) -> Result<()> {
135        // Rootless is currently not supported
136        if !getuid().is_root() {
137            bail!("Please run kubernix as root")
138        }
139
140        // Prepare the configuration
141        if config.root().exists() {
142            config.try_load_file()?;
143        } else {
144            config.to_file()?;
145        }
146        config.canonicalize_root()?;
147
148        // Setup the logger
149        set_boxed_logger(Logger::new(config.log_level(), config.log_format()))
150            .context("Unable to set logger")
151    }

153    /// Stop kubernix by cleaning up all running processes
154    fn stop(&mut self) {
155        for x in &mut self.processes {
156            if let Err(e) = x.stop() {
157                debug!("{}", e)
158            }
159        }
160    }

162    /// The amount of processes to be run
163    fn processes(config: &Config) -> u64 {
164        5 + 2 * u64::from(config.nodes())
165    }

167    /// Bootstrap the whole cluster, which assumes to be inside a nix shell
168    fn bootstrap_cluster(config: Config) -> Result<()> {
169        // Setup the progress bar
170        const BASE_STEPS: u64 = 15;
171        let steps = if config.multi_node() {
172            u64::from(config.nodes()) * 2 + BASE_STEPS
173        } else {
174            BASE_STEPS
175        } + Self::processes(&config);
176        let p = Progress::new(steps, config.log_level());
177        info!("Bootstrapping cluster");
178
179        // Ensure that the system is prepared
180        let system = System::setup(&config).context("Unable to setup system")?;
181        Container::build(&config)?;
182
183        // Setup the network
184        let network = Network::new(&config)?;
185
186        // Setup the public key infrastructure
187        let pki = Pki::new(&config, &network)?;
188
189        // Setup the configs
190        let kubeconfig = KubeConfig::new(&config, &pki)?;
191        let kubectl = Kubectl::new(kubeconfig.admin());
192        let encryptionconfig = EncryptionConfig::new(&config)?;
193
194        // Build the component registry
195        let ctx = ClusterContext {
196            config: &config,
197            network: &network,
198            pki: &pki,
199            kubeconfig: &kubeconfig,
200            encryptionconfig: &encryptionconfig,
201            kubectl: &kubectl,
202        };
203
204        let mut registry = ComponentRegistry::new();
205        registry.register(Box::new(etcd::EtcdComponent));
206        registry.register(Box::new(apiserver::ApiServerComponent));
207        registry.register(Box::new(controllermanager::ControllerManagerComponent));
208        registry.register(Box::new(scheduler::SchedulerComponent));
209        for node in 0..config.nodes() {
210            registry.register(Box::new(crio::CrioComponent::new(node)));
211            registry.register(Box::new(kubelet::KubeletComponent::new(node)));
212        }
213        registry.register(Box::new(proxy::ProxyComponent));
214
215        let (processes, all_ok) = registry.run(&ctx);
216
217        // Setup the main instance
218        let spawn_shell = !config.no_shell();
219        let addon_shutdown = Arc::new(AtomicBool::new(false));
220        let mut kubernix = Kubernix {
221            addon_shutdown: Arc::clone(&addon_shutdown),
222            addon_thread: None,
223            config,
224            network,
225            kubectl,
226            processes,
227            system,
228        };
229
230        // No dead processes
231        if all_ok {
232            if let Err(e) = kubernix.write_env_file() {
233                p.reset();
234                error!("Unable to write environment file: {}", e);
235                return Err(e);
236            }
237
238            // Deploy addons asynchronously so the shell is available
239            // sooner. Failures are logged but do not block startup.
240            // The thread handle is stored so Drop can join it before
241            // tearing down cluster processes.
242            let addon_config = kubernix.config.clone();
243            let addon_network = kubernix.network.clone();
244            let addon_kubeconfig = kubernix.kubectl.kubeconfig().to_path_buf();
245            kubernix.addon_thread = Some(thread::spawn(move || {
246                let kubectl = Kubectl::new(&addon_kubeconfig);
247                if addon_shutdown.load(Ordering::Relaxed) {
248                    return;
249                }
250                if addon_config.addons().iter().any(|a| a == "coredns") {
251                    if let Err(e) = CoreDns::apply(&addon_config, &addon_network, &kubectl) {
252                        if !addon_shutdown.load(Ordering::Relaxed) {
253                            error!("Failed to deploy CoreDNS addon: {}", e);
254                        }
255                    }
256                }
257            }));
258
259            info!("Everything is up and running");
260            p.reset();
261
262            if spawn_shell {
263                kubernix.spawn_shell()?;
264            } else {
265                kubernix.wait()?;
266            }
267        } else {
268            p.reset();
269            bail!("Unable to start all processes")
270        }
271
272        Ok(())
273    }

275    /// Wait until a termination signal occurs
276    fn wait(&self) -> Result<()> {
277        // Setup the signal handlers
278        let term = Arc::new(AtomicBool::new(false));
279        flag::register(SIGTERM, Arc::clone(&term))?;
280        flag::register(SIGINT, Arc::clone(&term))?;
281        flag::register(SIGHUP, Arc::clone(&term))?;
282        info!("Waiting for interrupt…");
283
284        // Write the pid file
285        let pid_file = self.config.root().join("kubernix.pid");
286        debug!("Writing pid file to: {}", pid_file.display());
287        fs::write(pid_file, id().to_string())?;
288
289        // Wait for the signals
290        while !term.load(Ordering::Relaxed) {
291            sleep(Duration::from_millis(100));
292        }
293        Ok(())
294    }

296    /// Spawn a new interactive default system shell
297    fn spawn_shell(&self) -> Result<()> {
298        info!("Spawning interactive shell");
299        info!("Please be aware that the cluster stops if you exit the shell");
300
301        let mut cmd = Command::new(self.config.shell_ok()?);
302        cmd.current_dir(self.config.root());
303        Self::apply_env_file(&Self::env_file(&self.config), &mut cmd)?;
304        cmd.status()?;
305        Ok(())
306    }

308    /// Parse the env file and apply its variables to a Command.
309    /// Handles `export KEY=VALUE` and `KEY=VALUE` formats, stripping
310    /// surrounding single or double quotes from values.
311    fn apply_env_file(env_file: &Path, cmd: &mut Command) -> Result<()> {
312        let content = fs::read_to_string(env_file)
313            .with_context(|| format!("Unable to read env file '{}'", env_file.display()))?;
314        for line in content.lines() {
315            let line = line.strip_prefix("export ").unwrap_or(line);
316            if let Some((key, value)) = line.split_once('=') {
317                let value = value
318                    .strip_prefix('"')
319                    .and_then(|v| v.strip_suffix('"'))
320                    .or_else(|| value.strip_prefix('\'').and_then(|v| v.strip_suffix('\'')))
321                    .unwrap_or(value);
322                cmd.env(key, value);
323            }
324        }
325        Ok(())
326    }
327
328    /// Lay out the env file
329    fn write_env_file(&self) -> Result<()> {
330        info!("Writing environment file");
331        fs::write(
332            Self::env_file(&self.config),
333            format!(
334                "export {}={}\nexport {}={}",
335                RUNTIME_ENV,
336                Crio::socket(&self.config, &self.network, 0)?.to_socket_string(),
337                "KUBECONFIG",
338                self.kubectl.kubeconfig().display(),
339            ),
340        )?;
341        Ok(())
342    }

344    /// Retrieve the path to the env file
345    fn env_file(config: &Config) -> PathBuf {
346        config.root().join("kubernix.env")
347    }

349    /// Remove all stale mounts
350    fn umount(&self) {
351        debug!("Removing active mounts");
352        let now = Instant::now();
353        while now.elapsed().as_secs() < 15 {
354            match Self::read_mount_points(self.config.root()) {
355                Err(e) => {
356                    debug!("Unable to retrieve mounts: {}", e);
357                    sleep(Duration::from_secs(1));
358                }
359                Ok(mount_points) => {
360                    if mount_points.is_empty() {
361                        break;
362                    }
363                    for dest in &mount_points {
364                        debug!("Removing mount: {}", dest.display());
365                        if let Err(e) = umount2(dest, MntFlags::MNT_FORCE) {
366                            debug!("Unable to umount '{}': {}", dest.display(), e);
367                        }
368                    }
369                    sleep(Duration::from_millis(500));
370                }
371            };
372        }
373    }

375    /// Read mount points from /proc/mounts filtered by the given root path,
376    /// sorted deepest-first for safe unmounting.
377    fn read_mount_points(root: &Path) -> Result<Vec<PathBuf>> {
378        let file = fs::File::open("/proc/mounts").context("Unable to open /proc/mounts")?;
379        let reader = BufReader::new(file);
380        let mut points: Vec<PathBuf> = reader
381            .lines()
382            .map_while(Result::ok)
383            .filter_map(|line| line.split_whitespace().nth(1).map(PathBuf::from))
384            .filter(|p| p.starts_with(root) && p != root)
385            .collect();
386        points.sort_by_key(|p| std::cmp::Reverse(p.components().count()));
387        Ok(points)
388    }

456impl Drop for Kubernix {
457    fn drop(&mut self) {
458        let p = Progress::new(Self::processes(&self.config), self.config.log_level());
459
460        info!("Cleaning up");
461
462        // Signal the addon thread to stop and give it a short window
463        // to finish. If it does not complete in time, proceed with
464        // shutdown rather than blocking for up to 120s.
465        self.addon_shutdown.store(true, Ordering::Relaxed);
466        if let Some(handle) = self.addon_thread.take() {
467            debug!("Waiting for addon deployment to finish");
468            let deadline = Instant::now() + Duration::from_secs(5);
469            while !handle.is_finished() && Instant::now() < deadline {
470                sleep(Duration::from_millis(100));
471            }
472            if handle.is_finished() {
473                let _ = handle.join();
474            } else {
475                debug!("Addon thread did not finish in time, proceeding with shutdown");
476            }
477        }
478
479        self.stop();
480        self.umount();
481        self.system.cleanup();
482        info!("Cleanup done");
483
484        p.reset();
485        debug!("All done");
486    }
