1//! Extensible component registry for cluster bootstrapping.
2//!
3//! Each cluster component (etcd, API server, etc.) implements the [`Component`]
4//! trait. Components declare their boot phase, which determines the order in
5//! which they are started. Within a phase, components start in parallel.
6
7use crate::{
8    config::Config,
9    encryptionconfig::EncryptionConfig,
10    kubeconfig::KubeConfig,
11    kubectl::Kubectl,
12    network::Network,
13    pki::Pki,
14    process::{ProcessState, Started, Stoppables},
15};
16use log::{debug, info};
17use rayon::prelude::*;
18
19/// Boot phases control the startup order of cluster components.
20/// Components in the same phase start concurrently; phases run sequentially.
21///
22/// Phase ordering is optimized so that independent components overlap:
23/// - CRI-O starts alongside controllers (it only needs etcd/apiserver, not controllers)
24/// - Proxy starts alongside kubelets (it only needs the API server to sync caches)
25#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord)]
26pub enum Phase {
27    /// Infrastructure services (etcd)
28    Infrastructure,
29    /// Control plane services (API server)
30    ControlPlane,
31    /// Controllers, node runtimes, and proxy all start together since they
32    /// only depend on the API server, not on each other.
33    /// Contains: scheduler, controller-manager, CRI-O (per node), proxy.
34    Controller,
35    /// Node agents that require a running CRI-O runtime (kubelet).
36    NodeAgent,
37}
38
39/// Shared cluster state passed to components during startup.
40pub struct ClusterContext<'a> {
41    /// Cluster configuration
42    pub config: &'a Config,
43    /// Network topology
44    pub network: &'a Network,
45    /// Public key infrastructure (certificates)
46    pub pki: &'a Pki,
47    /// Kubeconfig files for each component
48    pub kubeconfig: &'a KubeConfig,
49    /// Encryption configuration for the API server
50    pub encryptionconfig: &'a EncryptionConfig,
51    /// Kubectl handle for RBAC and workload operations
52    pub kubectl: &'a Kubectl,
53}
54
55/// Trait implemented by each cluster component.
56///
57/// This provides a uniform interface for the orchestrator to discover,
58/// start, and order components without hardcoding them in `lib.rs`.
59pub trait Component: Send + Sync {
60    /// Human-readable name for logging
61    fn name(&self) -> &str;
62
63    /// The boot phase this component belongs to
64    fn phase(&self) -> Phase;
65
66    /// Start the component. Returns a handle that can stop it.
67    fn start(&self, ctx: &ClusterContext<'_>) -> ProcessState;
68}
69
70/// A registry that collects components and groups them by phase.
71#[must_use]
72pub struct ComponentRegistry {
73    components: Vec<Box<dyn Component>>,
74}
75
76impl ComponentRegistry {
77    /// Create an empty registry
78    pub fn new() -> Self {
79        Self {
80            components: Vec::new(),
81        }
82    }
83
84    /// Register a component
85    pub fn register(&mut self, component: Box<dyn Component>) {
86        self.components.push(component);
87    }
88
89    /// Return components grouped and sorted by phase
90    pub fn by_phase(&self) -> Vec<(Phase, Vec<&dyn Component>)> {
91        let mut phases: Vec<Phase> = self.components.iter().map(|c| c.phase()).collect();
92        phases.sort();
93        phases.dedup();
94
95        phases
96            .into_iter()
97            .map(|phase| {
98                let group: Vec<&dyn Component> = self
99                    .components
100                    .iter()
101                    .filter(|c| c.phase() == phase)
102                    .map(|c| c.as_ref())
103                    .collect();
104                (phase, group)
105            })
106            .collect()
107    }
108
109    /// Execute all registered components phase by phase.
110    ///
111    /// Within each phase, components start in parallel via rayon.
112    /// If any component in a phase fails, subsequent phases are skipped
113    /// (later phases depend on earlier ones).
114    ///
115    /// The Controller phase runs controllers, CRI-O, and proxy in parallel
116    /// since they all only depend on the API server.
117    ///
118    /// Returns stoppable handles in reverse phase order (for proper
119    /// shutdown: node agents before runtimes, runtimes before control
120    /// plane, etc.) and a flag indicating whether all succeeded.
121    pub fn run(&self, ctx: &ClusterContext<'_>) -> (Stoppables, bool) {
122        info!("Starting processes");
123        let mut all_results: Vec<(Phase, ProcessState)> = Vec::new();
124
125        for (phase, components) in &self.by_phase() {
126            debug!(
127                "Starting {:?} phase ({} components)",
128                phase,
129                components.len()
130            );
131            let results: Vec<ProcessState> = components
132                .par_iter()
133                .map(|c| {
134                    debug!("Starting {}", c.name());
135                    c.start(ctx)
136                })
137                .collect();
138
139            let phase_ok = results.iter().all(|r| r.is_ok());
140            all_results.extend(results.into_iter().map(|r| (*phase, r)));
141
142            if !phase_ok {
143                info!("Phase {:?} failed, skipping remaining phases", phase);
144                break;
145            }
146        }
147
148        let all_ok = all_results.iter().all(|(_, r)| r.is_ok());
149
150        // Collect successful starts, then reverse so shutdown happens
151        // in reverse phase order.
152        let mut stoppables: Vec<(Phase, Started)> = all_results
153            .into_iter()
154            .filter_map(|(phase, r)| match r {
155                Ok(p) => Some((phase, p)),
156                Err(e) => {
157                    debug!("{}", e);
158                    None
159                }
160            })
161            .collect();
162        stoppables.reverse();
163
164        let processes: Stoppables = stoppables.into_iter().map(|(_, p)| p).collect();
165        (processes, all_ok)
166    }
167}

169#[cfg(test)]
170mod tests {
171    use super::*;
172    use anyhow::bail;
173
174    struct FakeComponent {
175        name: String,
176        phase: Phase,
177    }
178
179    impl Component for FakeComponent {
180        fn name(&self) -> &str {
181            &self.name
182        }
183
184        fn phase(&self) -> Phase {
185            self.phase
186        }
187
188        fn start(&self, _ctx: &ClusterContext<'_>) -> ProcessState {
189            bail!("fake component")
190        }
191    }
192
193    #[test]
194    fn registry_groups_by_phase() {
195        let mut reg = ComponentRegistry::new();
196        reg.register(Box::new(FakeComponent {
197            name: "etcd".into(),
198            phase: Phase::Infrastructure,
199        }));
200        reg.register(Box::new(FakeComponent {
201            name: "apiserver".into(),
202            phase: Phase::ControlPlane,
203        }));
204        reg.register(Box::new(FakeComponent {
205            name: "scheduler".into(),
206            phase: Phase::Controller,
207        }));
208        reg.register(Box::new(FakeComponent {
209            name: "controller-manager".into(),
210            phase: Phase::Controller,
211        }));
212
213        let phases = reg.by_phase();
214        assert_eq!(phases.len(), 3);
215        assert_eq!(phases[0].0, Phase::Infrastructure);
216        assert_eq!(phases[0].1.len(), 1);
217        assert_eq!(phases[1].0, Phase::ControlPlane);
218        assert_eq!(phases[1].1.len(), 1);
219        assert_eq!(phases[2].0, Phase::Controller);
220        assert_eq!(phases[2].1.len(), 2);
221    }
222
223    #[test]
224    fn empty_registry() {
225        let reg = ComponentRegistry::new();
226        assert!(reg.by_phase().is_empty());
227    }
