1//! Generic process lifecycle management.
2//!
3//! Provides the [`Process`] abstraction that spawns a child process,
4//! watches it for unexpected exits, waits for readiness based on log
5//! output patterns, and handles graceful shutdown via SIGTERM.
6

25/// A general process abstraction
26pub struct Process {
27    command: String,
28    died: Receiver<()>,
29    kill: Sender<()>,
30    log_file: PathBuf,
31    name: String,
32    pid: u32,
33    readiness_timeout: u64,
34    watch: Option<JoinHandle<Result<()>>>,
35}
36
37/// The trait to stop something
38pub trait Stoppable {
39    /// Stop the process
40    fn stop(&mut self) -> Result<()>;
41}
42
43/// A running process handle that can be stopped during cluster shutdown.
44pub type Started = Box<dyn Stoppable + Send + Sync>;
45
46/// Ordered collection of running processes, stopped in reverse order during cleanup.
47pub type Stoppables = Vec<Started>;
48
49/// The result of starting a component process, either a stoppable handle or an error.
50pub type ProcessState = Result<Started>;
51
52#[derive(Deserialize, Serialize)]
53struct Run {
54    command: PathBuf,
55    args: Vec<String>,
56}
57
58impl Process {
59    /// Creates a new `Process` instance by spawning the provided `command` and `args`.
60    /// If the process creation fails, an `Error` will be returned.
61    pub fn start(dir: &Path, identifier: &str, command: &str, args: &[&str]) -> Result<Process> {
62        // Prepare the commands
63        if command.is_empty() {
64            bail!("No valid command provided")
65        }
66        info!("Starting {}", identifier);
67
68        // Write the executed command into the dir
69        create_dir_all(dir)?;
70
71        // If the run file exists, execute only that one
72        let run_file = dir.join("run.json");
73        let run = if !run_file.exists() {
74            debug!(
75                "No previous run file '{}' found, writing new one",
76                run_file.display()
77            );
78            // Write the run file
79            let f = Run {
80                command: System::find_executable(command)?,
81                args: args.iter().map(|x| (*x).to_string()).collect(),
82            };
83            fs::write(&run_file, serde_json::to_string_pretty(&f)?)?;
84            f
85        } else {
86            debug!("Re using run file '{}'", run_file.display());
87            let contents = fs::read_to_string(&run_file)?;
88            serde_json::from_str(&contents)?
89        };
90
91        // Prepare the log dir and file
92        let mut log_file = dir.join(command);
93        log_file.set_extension("log");
94        let out_file = File::create(&log_file)?;
95        let err_file = out_file.try_clone()?;
96
97        // Spawn the process child
98        let mut child = Command::new(run.command)
99            .args(run.args)
100            .stderr(Stdio::from(err_file))
101            .stdout(Stdio::from(out_file))
102            .spawn()
103            .with_context(|| format!("Unable to start process '{}' ({})", identifier, command,))?;
104
105        // Start the watcher thread
106        let (kill, killed) = bounded(1);
107        let (dead, died) = bounded(1);
108        let c = command.to_owned();
109        let n = identifier.to_owned();
110        let pid = child.id();
111        let lf = log_file.clone();
112        let watch = spawn(move || {
113            // Wait for the process to exit
114            let status = child.wait()?;
115
116            // No kill send, we assume that the process died
117            if killed.try_recv().is_err() {
118                error!("{} ({}) died unexpectedly", n, c);
119                Self::dump_log_tail(&lf, &n);
120                dead.send(())?;
121            } else {
122                info!("{} stopped", n);
123            }
124            debug!("{} ({}) {}", n, c, status);
125            Ok(())
126        });
127
128        Ok(Process {
129            command: command.into(),
130            died,
131            kill,
132            log_file,
133            name: identifier.into(),
134            pid,
135            readiness_timeout: 120,
136            watch: Some(watch),
137        })
138    }
139
140    /// Wait for the process to become ready, by searching for the pattern in
141    /// every line of its output.
142    pub fn wait_ready(&mut self, pattern: &str) -> Result<()> {
143        debug!(
144            "Waiting for process '{}' ({}) to become ready with pattern: '{}'",
145            self.name, self.command, pattern
146        );
147        let now = Instant::now();
148        let file = File::open(&self.log_file)?;
149        let mut reader = BufReader::new(file);
150        let mut line = String::new();
151
152        while now.elapsed().as_secs() < self.readiness_timeout {
153            line.clear();
154            reader.read_line(&mut line)?;
155
156            if line.is_empty() {
157                if self.died.try_recv().is_ok() {
158                    bail!(
159                        "{} ({}) died before becoming ready",
160                        self.name,
161                        self.command
162                    )
163                }
164                sleep(Duration::from_millis(10));
165                continue;
166            }
167
168            if line.contains(pattern) {
169                info!("{} is ready", self.name);
170                debug!("Found pattern '{}' in line '{}'", pattern, line.trim());
171                return Ok(());
172            }
173        }
174
175        // Dump log before stopping so the file is still intact
176        Self::dump_log_tail(&self.log_file, &self.name);
177        self.stop()?;
178        bail!(
179            "Timed out after {}s waiting for process '{}' ({}) to become ready \
180             with pattern '{}' (log: {})",
181            self.readiness_timeout,
182            self.name,
183            self.command,
184            pattern,
185            self.log_file.display(),
186        )
187    }
188
189    /// Dump the last lines of a process log file for debugging
190    fn dump_log_tail(log_file: &Path, name: &str) {
191        if let Ok(content) = fs::read_to_string(log_file) {
192            let lines: Vec<&str> = content.lines().collect();
193            let start = lines.len().saturating_sub(20);
194            error!("Last log lines of {}:", name);
195            for line in &lines[start..] {
196                error!("  {}", line);
197            }
198        }
199    }
200}
201
202impl Stoppable for Process {
203    /// Stopping the process by killing it
204    fn stop(&mut self) -> Result<()> {
205        debug!("Stopping process {} (via {})", self.name, self.command);
206
207        // Indicate that this shutdown is intended
208        self.kill.send(()).with_context(|| {
209            format!(
210                "Unable to send kill signal to process {} (via {})",
211                self.name, self.command,
212            )
213        })?;
214
215        // Send SIGTERM to the process
216        let pid = i32::try_from(self.pid)
217            .with_context(|| format!("PID {} exceeds i32 range", self.pid))?;
218        kill(Pid::from_raw(pid), Signal::SIGTERM)?;
219
220        // Join the waiting thread
221        if let Some(handle) = self.watch.take()
222            && handle.join().is_err()
223        {
224            bail!(
225                "Unable to stop process {} (via {})",
226                self.name,
227                self.command
228            );
229        }
230        debug!("Process {} (via {}) stopped", self.name, self.command);
231        Ok(())
232    }
233}
