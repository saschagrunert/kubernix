1//! System preparation and host utilities.
2//!
3//! Handles kernel module loading, sysctl configuration, hosts file
4//! management, executable lookup in `$PATH`, and container detection.
5
6use crate::{Config, node::Node};
7use anyhow::{Context, Result, bail};
8use log::{debug, info, warn};
9use std::{
10    env::{split_paths, var, var_os},
11    fmt::Display,
12    fs::{self, read_to_string},
13    net::Ipv4Addr,
14    path::{Path, PathBuf},
15    process::Command,
16};
17
18/// Manages host-level system preparation and cleanup.
19pub struct System {
20    hosts: Option<String>,
21}
22
23impl System {
24    /// Create a new system
25    pub fn setup(config: &Config) -> Result<Self> {
26        if Self::in_container()? {
27            info!("Skipping modprobe and sysctl for sake of containerization")
28        } else {
29            for module in &["overlay", "br_netfilter", "ip_conntrack"] {
30                Self::modprobe(module)?;
31            }
32            for sysctl in &[
33                "net.bridge.bridge-nf-call-ip6tables",
34                "net.bridge.bridge-nf-call-iptables",
35                "net.ipv4.conf.all.route_localnet",
36                "net.ipv4.ip_forward",
37            ] {
38                Self::sysctl_enable(sysctl)?;
39            }
40        }
41
42        let hosts = if config.multi_node() {
43            // Try to write the hostnames, which does not work on every system
44            let hosts_file = Self::hosts();
45            let hosts = read_to_string(&hosts_file)?;
46            let local_hosts = (0..config.nodes())
47                .map(|x| format!("{} {}", Ipv4Addr::LOCALHOST, Node::raw(x)))
48                .collect::<Vec<_>>();
49
50            let mut new_hosts = hosts
51                .lines()
52                .filter(|x| !local_hosts.iter().any(|y| x == y))
53                .map(|x| x.into())
54                .collect::<Vec<_>>();
55            new_hosts.extend(local_hosts);
56
57            match fs::write(&hosts_file, new_hosts.join("\n")) {
58                Err(e) => {
59                    warn!(
60                        "Unable to write hosts file '{}'. The nodes may be not reachable: {}",
61                        hosts_file.display(),
62                        e
63                    );
64                    None
65                }
66                _ => Some(hosts),
67            }
68        } else {
69            None
70        };
71
72        Ok(Self { hosts })
73    }
74
75    /// Returns true if the process is running inside a container
76    pub fn in_container() -> Result<bool> {
77        Ok(
78            read_to_string(PathBuf::from("/").join("proc").join("1").join("cgroup"))
79                .context("Unable to retrieve systems container status")?
80                .lines()
81                .any(|x| x.contains("libpod") || x.contains("podman") || x.contains("docker")),
82        )
83    }
84
85    /// Restore the initial system state
86    pub fn cleanup(&self) {
87        if let Some(hosts) = &self.hosts
88            && let Err(e) = fs::write(Self::hosts(), hosts)
89        {
90            warn!(
91                "Unable to restore hosts file, may need manual cleanup: {}",
92                e
93            )
94        }
95    }
96
97    /// Find an executable inside the current $PATH environment
98    pub fn find_executable<P>(name: P) -> Result<PathBuf>
99    where
100        P: AsRef<Path> + Display,
101    {
102        var_os("PATH")
103            .and_then(|paths| {
104                split_paths(&paths)
105                    .filter_map(|dir| {
106                        let full_path = dir.join(&name);
107                        if full_path.is_file() {
108                            Some(full_path)
109                        } else {
110                            None
111                        }
112                    })
113                    .next()
114            })
115            .with_context(|| format!("Unable to find executable '{}' in $PATH", name))
116    }
117
118    /// Return the full path to the default system shell
119    pub fn shell() -> Result<String> {
120        let shell = var("SHELL").unwrap_or_else(|_| "sh".into());
121        Ok(format!(
122            "{}",
123            Self::find_executable(&shell)
124                .with_context(|| format!("Unable to find system shell '{}'", shell))?
125                .display()
126        ))
127    }
128
129    /// Load a single kernel module via 'modprobe'
130    fn modprobe(module: &str) -> Result<()> {
131        debug!("Loading kernel module '{}'", module);
132        let output = Command::new("modprobe").arg(module).output()?;
133        if !output.status.success() {
134            bail!(
135                "Unable to load '{}' kernel module: {}",
136                module,
137                String::from_utf8(output.stderr)?,
138            );
139        }
140        Ok(())
141    }
142
143    /// Enable a single sysctl by setting it to '1'
144    fn sysctl_enable(key: &str) -> Result<()> {
145        debug!("Enabling sysctl '{}'", key);
146        let enable_arg = format!("{}=1", key);
147        let output = Command::new("sysctl").arg("-w").arg(&enable_arg).output()?;
148        let stderr = String::from_utf8(output.stderr)?;
149        if !stderr.is_empty() {
150            bail!("Unable to set sysctl '{}': {}", enable_arg, stderr);
151        }
152        Ok(())
153    }
154
155    fn hosts() -> PathBuf {
156        PathBuf::from("/").join("etc").join("hosts")
157    }
158}
