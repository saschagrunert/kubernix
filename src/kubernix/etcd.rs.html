1//! etcd distributed key-value store component.
2//!
3//! Provides the backing store for all Kubernetes cluster data.
4//! Starts a single-node etcd instance with TLS mutual authentication.
5
6use crate::{
7    component::{ClusterContext, Component, Phase},
8    config::Config,
9    network::Network,
10    pki::Pki,
11    process::{Process, ProcessState, Stoppable},
12};
13use anyhow::Result;
14use std::fs::create_dir_all;
15
16/// Component wrapper for registry-based startup.
17pub struct EtcdComponent;
18
19impl Component for EtcdComponent {
20    fn name(&self) -> &str {
21        "etcd"
22    }
23
24    fn phase(&self) -> Phase {
25        Phase::Infrastructure
26    }
27
28    fn start(&self, ctx: &ClusterContext<'_>) -> ProcessState {
29        Etcd::start(ctx.config, ctx.network, ctx.pki)
30    }
31}
32
33/// Manages the etcd process lifecycle.
34pub struct Etcd {
35    process: Process,
36}
37
38impl Etcd {
39    /// Start etcd with TLS and the given cluster configuration.
40    pub fn start(config: &Config, network: &Network, pki: &Pki) -> ProcessState {
41        const ETCD: &str = "etcd";
42        let dir = config.root().join(ETCD);
43        create_dir_all(&dir)?;
44
45        let mut process = Process::start(
46            &dir,
47            ETCD,
48            ETCD,
49            &[
50                "--auto-compaction-mode=periodic",
51                "--auto-compaction-retention=1h",
52                "--client-cert-auth",
53                "--initial-cluster-state=new",
54                "--initial-cluster-token=etcd-cluster",
55                "--peer-client-cert-auth",
56                "--snapshot-count=5000",
57                &format!(
58                    "--initial-advertise-peer-urls=https://{}",
59                    network.etcd_peer()
60                ),
61                &format!("--advertise-client-urls=https://{}", network.etcd_client()),
62                &format!("--cert-file={}", pki.apiserver().cert().display()),
63                &format!("--data-dir={}", dir.join("run").display()),
64                &format!("--initial-cluster=etcd=https://{}", network.etcd_peer()),
65                &format!("--key-file={}", pki.apiserver().key().display()),
66                &format!("--listen-client-urls=https://{}", network.etcd_client()),
67                &format!("--listen-peer-urls=https://{}", network.etcd_peer()),
68                &format!("--name={}", ETCD),
69                &format!("--peer-cert-file={}", pki.apiserver().cert().display()),
70                &format!("--peer-key-file={}", pki.apiserver().key().display()),
71                &format!("--peer-trusted-ca-file={}", pki.ca().cert().display()),
72                &format!("--trusted-ca-file={}", pki.ca().cert().display()),
73            ],
74        )?;
75
76        process.wait_ready("ready to serve client requests")?;
77        Ok(Box::new(Self { process }))
78    }
79}
80
81impl Stoppable for Etcd {
82    fn stop(&mut self) -> Result<()> {
83        self.process.stop()
84    }
85}
