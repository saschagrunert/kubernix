1//! Kubernetes API server component.
2//!
3//! Starts `kube-apiserver` with TLS, etcd connectivity, RBAC authorization,
4//! and encryption-at-rest, then applies the kubelet API RBAC rules.
5
6use crate::{
7    component::{ClusterContext, Component, Phase},
8    config::Config,
9    encryptionconfig::EncryptionConfig,
10    kubectl::Kubectl,
11    network::Network,
12    pki::Pki,
13    process::{Process, ProcessState, Stoppable},
14};
15use anyhow::{Context, Result};
16use log::debug;
17use std::{
18    fs::{self, create_dir_all},
19    path::Path,
20};
21
22/// Component wrapper for registry-based startup.
23pub struct ApiServerComponent;
24
25impl Component for ApiServerComponent {
26    fn name(&self) -> &str {
27        "API Server"
28    }
29
30    fn phase(&self) -> Phase {
31        Phase::ControlPlane
32    }
33
34    fn start(&self, ctx: &ClusterContext<'_>) -> ProcessState {
35        ApiServer::start(
36            ctx.config,
37            ctx.network,
38            ctx.pki,
39            ctx.encryptionconfig,
40            ctx.kubectl,
41        )
42    }
43}
44
45/// Manages the `kube-apiserver` process lifecycle.
46pub struct ApiServer {
47    process: Process,
48}
49
50impl ApiServer {
51    /// Start the API server with the given cluster configuration.
52    pub fn start(
53        config: &Config,
54        network: &Network,
55        pki: &Pki,
56        encryptionconfig: &EncryptionConfig,
57        kubectl: &Kubectl,
58    ) -> ProcessState {
59        let dir = config.root().join("apiserver");
60        create_dir_all(&dir)?;
61
62        let mut process = Process::start(
63            &dir,
64            "API Server",
65            "kube-apiserver",
66            &[
67                "--audit-log-maxage=30",
68                "--audit-log-maxbackup=3",
69                "--audit-log-maxsize=100",
70                &format!("--audit-log-path={}", dir.join("audit.log").display()),
71                "--authorization-mode=Node,RBAC",
72                "--bind-address=0.0.0.0",
73                &format!("--client-ca-file={}", pki.ca().cert().display()),
74                &format!("--etcd-cafile={}", pki.ca().cert().display()),
75                &format!("--etcd-certfile={}", pki.apiserver().cert().display()),
76                &format!("--etcd-keyfile={}", pki.apiserver().key().display()),
77                &format!("--etcd-servers=https://{}", network.etcd_client()),
78                "--event-ttl=1h",
79                &format!(
80                    "--encryption-provider-config={}",
81                    encryptionconfig.path().display()
82                ),
83                &format!(
84                    "--kubelet-certificate-authority={}",
85                    pki.ca().cert().display()
86                ),
87                &format!(
88                    "--kubelet-client-certificate={}",
89                    pki.apiserver().cert().display()
90                ),
91                &format!("--kubelet-client-key={}", pki.apiserver().key().display()),
92                "--runtime-config=api/all=true",
93                "--service-account-issuer=https://kubernetes.default.svc.cluster.local",
94                &format!(
95                    "--service-account-key-file={}",
96                    pki.service_account().cert().display()
97                ),
98                &format!(
99                    "--service-account-signing-key-file={}",
100                    pki.service_account().key().display()
101                ),
102                &format!("--service-cluster-ip-range={}", network.service_cidr()),
103                &format!("--tls-cert-file={}", pki.apiserver().cert().display()),
104                &format!("--tls-private-key-file={}", pki.apiserver().key().display()),
105                "--v=2",
106            ],
107        )?;
108
109        process.wait_ready("Serving securely")?;
110        Self::setup_rbac(&dir, kubectl)?;
111        Ok(Box::new(Self { process }))
112    }
113
114    fn setup_rbac(dir: &Path, kubectl: &Kubectl) -> Result<()> {
115        debug!("Creating API Server RBAC rule for kubelet");
116        let file = dir.join("rbac.yml");
117
118        if !file.exists() {
119            fs::write(&file, include_str!("assets/apiserver.yml"))?;
120        }
121
122        kubectl
123            .apply(&file)
124            .context("Unable to deploy API server RBAC rules")?;
125
126        debug!("API Server RBAC rule created");
127        Ok(())
128    }
129}
130
131impl Stoppable for ApiServer {
132    fn stop(&mut self) -> Result<()> {
133        self.process.stop()
134    }
135}
