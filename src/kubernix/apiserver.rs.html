1use crate::{

4    kubectl::Kubectl,

9use anyhow::{Context, Result};

11use std::{

13    path::Path,
14};
15
16pub struct ApiServer {
17    process: Process,
18}
19
20impl ApiServer {
21    pub fn start(
22        config: &Config,
23        network: &Network,
24        pki: &Pki,
25        encryptionconfig: &EncryptionConfig,
26        kubectl: &Kubectl,
27    ) -> ProcessState {
28        let dir = config.root().join("apiserver");
29        create_dir_all(&dir)?;
30
31        let mut process = Process::start(
32            &dir,
33            "API Server",
34            "kube-apiserver",
35            &[
36                "--audit-log-maxage=30",
37                "--audit-log-maxbackup=3",
38                "--audit-log-maxsize=100",
39                &format!("--audit-log-path={}", dir.join("audit.log").display()),
40                "--authorization-mode=Node,RBAC",
41                "--bind-address=0.0.0.0",
42                &format!("--client-ca-file={}", pki.ca().cert().display()),
43                &format!("--etcd-cafile={}", pki.ca().cert().display()),
44                &format!("--etcd-certfile={}", pki.apiserver().cert().display()),
45                &format!("--etcd-keyfile={}", pki.apiserver().key().display()),
46                &format!("--etcd-servers=https://{}", network.etcd_client()),
47                "--event-ttl=1h",
48                &format!(
49                    "--encryption-provider-config={}",
50                    encryptionconfig.path().display()
51                ),
52                &format!(
53                    "--kubelet-certificate-authority={}",
54                    pki.ca().cert().display()
55                ),
56                &format!(
57                    "--kubelet-client-certificate={}",
58                    pki.apiserver().cert().display()
59                ),
60                &format!("--kubelet-client-key={}", pki.apiserver().key().display()),
61                "--runtime-config=api/all=true",
62                "--service-account-issuer=https://kubernetes.default.svc.cluster.local",
63                &format!(
64                    "--service-account-key-file={}",
65                    pki.service_account().cert().display()
66                ),
67                &format!(
68                    "--service-account-signing-key-file={}",
69                    pki.service_account().key().display()
70                ),
71                &format!("--service-cluster-ip-range={}", network.service_cidr()),
72                &format!("--tls-cert-file={}", pki.apiserver().cert().display()),
73                &format!("--tls-private-key-file={}", pki.apiserver().key().display()),
74                "--v=2",
75            ],
76        )?;
77
78        process.wait_ready("Serving securely")?;
79        Self::setup_rbac(&dir, kubectl)?;
80        Ok(Box::new(Self { process }))
81    }
82
83    fn setup_rbac(dir: &Path, kubectl: &Kubectl) -> Result<()> {
84        debug!("Creating API Server RBAC rule for kubelet");
85        let file = dir.join("rbac.yml");
86
87        if !file.exists() {
88            fs::write(&file, include_str!("assets/apiserver.yml"))?;
89        }
90
91        kubectl
92            .apply(&file)
93            .context("Unable to deploy API server RBAC rules")?;
94
95        debug!("API Server RBAC rule created");
96        Ok(())
97    }
98}
99
100impl Stoppable for ApiServer {
101    fn stop(&mut self) -> Result<()> {
102        self.process.stop()
103    }
104}
