1//! Nix environment bootstrapping.
2//!
3//! Manages the `nix develop` shell that provides all runtime
4//! dependencies (etcd, kubernetes, CRI-O, etc.) at pinned versions.
5//! Re-executes the kubernix binary inside the Nix shell, forwarding
6//! all CLI options.
7
8use crate::{Config, system::System};
9use anyhow::{Result, bail};
10use log::{debug, info};
11use std::{
12    env::{current_exe, var},
13    fs::{self, create_dir_all},
14    process::Command,
15};
16
17/// Handles Nix shell bootstrapping and command execution.
18pub struct Nix;
19
20impl Nix {
21    pub const DIR: &'static str = "nix";
22    const NIX_ENV: &'static str = "IN_NIX";
23
24    /// Bootstrap the nix environment
25    pub fn bootstrap(config: Config) -> Result<()> {
26        // Prepare the nix dir
27        debug!("Nix environment not found, bootstrapping one");
28        let dir = config.root().join(Self::DIR);
29
30        // Write the configuration if not existing
31        if !dir.exists() {
32            create_dir_all(&dir)?;
33
34            fs::write(
35                dir.join("flake.nix"),
36                include_str!("../nix/runtime-flake.nix")
37                    .replace("KUBERNIX_SYSTEM", Self::nix_system()?),
38            )?;
39            fs::write(dir.join("flake.lock"), include_str!("../flake.lock"))?;
40
41            let packages = &config.packages().join(" ");
42            debug!("Adding additional packages: {:?}", config.packages());
43            fs::write(
44                dir.join("packages.nix"),
45                include_str!("../nix/packages.nix").replace("/* PACKAGES */", packages),
46            )?;
47
48            // Apply the overlay if existing
49            let target_overlay = dir.join("overlay.nix");
50            match config.overlay() {
51                // User defined overlay
52                Some(overlay) => {
53                    info!("Using custom overlay '{}'", overlay.display());
54                    fs::copy(overlay, target_overlay)?;
55                }
56
57                // The default overlay
58                None => {
59                    debug!("Using default overlay");
60                    fs::write(target_overlay, include_str!("../nix/overlay.nix"))?;
61                }
62            }
63
64            // Initialize a standalone git repo so that Nix flakes can
65            // discover the generated files. Without this, Nix would use
66            // the parent git worktree and filter to only tracked files.
67            Self::git_init(&dir)?;
68        }
69
70        // Run the shell, forwarding all config options
71        let exe = format!("{}", current_exe()?.display());
72        let root = format!("{}", config.root().display());
73        let log_level = format!("{}", config.log_level());
74        let log_format = format!("{}", config.log_format());
75        let cidr = format!("{}", config.cidr());
76        let nodes = format!("{}", config.nodes());
77        let container_runtime = config.container_runtime();
78
79        let shell_val: String = config.shell().unwrap_or_default().to_owned();
80        let overlay_val = config.overlay().map(|o| format!("{}", o.display()));
81        let mut args = vec![
82            exe.as_str(),
83            "--root",
84            root.as_str(),
85            "--log-level",
86            log_level.as_str(),
87            "--log-format",
88            log_format.as_str(),
89            "--cidr",
90            cidr.as_str(),
91            "--nodes",
92            nodes.as_str(),
93            "--container-runtime",
94            container_runtime,
95        ];
96
97        if let Some(ref overlay) = overlay_val {
98            args.push("--overlay");
99            args.push(overlay.as_str());
100        }
101
102        let dockerfile_val = config.dockerfile().map(|d| format!("{}", d.display()));
103        if let Some(ref dockerfile) = dockerfile_val {
104            args.push("--dockerfile");
105            args.push(dockerfile.as_str());
106        }
107
108        for pkg in config.packages() {
109            args.push("--packages");
110            args.push(pkg.as_str());
111        }
112
113        for addon in config.addons() {
114            args.push("--addons");
115            args.push(addon.as_str());
116        }
117
118        if config.no_shell() {
119            args.push("--no-shell");
120        } else if !shell_val.is_empty() {
121            args.push("--shell");
122            args.push(&shell_val);
123        }
124
125        Self::run(&config, &args)
126    }

128    /// Run a command inside the nix develop shell
129    pub fn run(config: &Config, args: &[&str]) -> Result<()> {
130        let nix_dir = config.root().join(Self::DIR);
131        let flake_ref = format!("path:{}", nix_dir.display());
132
133        let status = Command::new(System::find_executable("nix")?)
134            .env(Nix::NIX_ENV, "true")
135            .arg("develop")
136            .arg(&flake_ref)
137            .arg("--no-update-lock-file")
138            .arg("--command")
139            .arg("bash")
140            .arg("-c")
141            .arg(args.join(" "))
142            .status()?;
143        if !status.success() {
144            bail!("nix develop exited with status {}", status);
145        }
146        Ok(())
147    }

149    /// Initialize a git repo in the given directory so Nix flakes
150    /// treats it as a standalone source tree.
151    fn git_init(dir: &std::path::Path) -> Result<()> {
152        let git = System::find_executable("git")?;
153        let run = |args: &[&str]| -> Result<()> {
154            let status = Command::new(&git).arg("-C").arg(dir).args(args).status()?;
155            if !status.success() {
156                bail!("git {} failed with status {}", args.join(" "), status,);
157            }
158            Ok(())
159        };
160        run(&["init", "-q"])?;
161        run(&["config", "user.email", "kubernix@localhost"])?;
162        run(&["config", "user.name", "kubernix"])?;
163        run(&["config", "commit.gpgsign", "false"])?;
164        run(&["add", "."])?;
165        run(&["commit", "-q", "-m", "init", "--allow-empty"])?;
166        Ok(())
167    }

169    /// Returns the Nix system string for the current architecture.
170    fn nix_system() -> Result<&'static str> {
171        match std::env::consts::ARCH {
172            "x86_64" => Ok("x86_64-linux"),
173            "aarch64" => Ok("aarch64-linux"),
174            arch => bail!(
175                "unsupported architecture '{}' (only x86_64 and aarch64 Linux are supported)",
176                arch,
177            ),
178        }
179    }

181    /// Returns true if running in nix environment
182    pub fn is_active() -> bool {
183        var(Nix::NIX_ENV).is_ok()
184    }
185
186    /// Set the NIX_ENV marker so re-exec knows we are inside Nix.
187    ///
188    /// # Safety
189    /// Callers must ensure no other threads are concurrently reading
190    /// or writing environment variables.
191    #[cfg(test)]
192    pub fn set_env_marker() {
193        unsafe { std::env::set_var(Self::NIX_ENV, "true") };
194    }
195
196    /// Remove the NIX_ENV marker.
197    ///
198    /// # Safety
199    /// Callers must ensure no other threads are concurrently reading
200    /// or writing environment variables.
201    #[cfg(test)]
202    pub fn remove_env_marker() {
203        unsafe { std::env::remove_var(Self::NIX_ENV) };
204    }

213    #[test]
214    fn nix_system_success() {
215        let system = Nix::nix_system().unwrap();
216        assert!(
217            system == "x86_64-linux" || system == "aarch64-linux",
218            "unexpected system: {}",
219            system,
220        );
221        assert!(system.ends_with("-linux"));
222    }

225    fn git_init_success() {
226        let dir = tempdir().unwrap().keep();
227        std::fs::write(dir.join("test.txt"), "hello").unwrap();
228        Nix::git_init(&dir).unwrap();
229        assert!(dir.join(".git").exists());
230    }
